[Route("api/[controller]")]
public class TodoController:Controller
{
	[HttpGet]
	public IEnumerable<TodoItem> GetAll()
	{
	}

	[HttpGet("{id}", Name = "GetTodo")]
	public IActionResult GetById(string id)
	{
	}
	[HttpPost]
	public IActionResult Create([FromBody] TodoItem item)
	{
		if (item == null)
		{
			return BadRequest();
		}
		TodoItems.Add(item);
		return CreatedAtRoute("GetTodo", new { id = item.Key }, item);
	}
}

1) 首先解析controller中的 api/[controller]  把 [controller] 换成 todo (不区分大小写) , url变成了 /api/todo
2) 再解析每个方法,
	没有 template(第一个方法) ,get /api/todo/
	有template(第二个方法), 则把template添加到url的后面 get /api/todo/{id} ,还创建了一个命名的 route (GetTodo)
	第三个方法 post /api/todo/ ,CreatedAtRoute返回一个201的状态码,表明服务器创建了新的资源,前2个参数用于生成response的Location,第三个参数是返回对象
