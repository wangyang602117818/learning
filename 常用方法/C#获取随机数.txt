public string GetRandomNum(int numb, char model = 'a')
{
    Random r = new Random();
    StringBuilder sb = new StringBuilder();
    for (int i = 0; i < numb; i++)
    {
	int res = r.Next(0, 3); 
	switch (res)
	{
	    case 0:
		sb.Append(r.Next(0, 10));  //����
		break;
	    case 1:
		sb.Append((char)r.Next(97, 123));  //Сд��ĸ
		break;
	    case 2:
		sb.Append((char)r.Next(65, 91));  //��д��ĸ
		break;
	}
    }
    return char.IsLower(model) ? sb.ToString().ToLower() : sb.ToString().ToUpper();
}
